function bgChage(){
	var date = new Date();
	var hour = date.getHours();

	if( hour >= 0 && hour < 16){
		setcolor = 'yellow';
		alert("Good Mornig !!")
	}else if( hour >= 16 && hour < 19){
		setcolor = 'blue';
		alert("Good Evening !!")
	}else{
		setcolor = 'black';
		alert("Good Night !!")
	}

	document.getElementsByTagName('body')[0].style.backgroundColor = setcolor;
}
